def shape(M):

        if not M:
            raise ValueError
        cols = len(M[0])
        rows = len(M)

        raise TypeError('The coordinates must be an iterable')

def matxMultiply(A, B):
    try:
        rA, cA = shape(A)
        rB, cB = shape(B)
        if cA != rB:
            raise ValueError
        M = []
        for i in range(rA):
            M.append([])
            for j in range(cB):
                temp = []
                for k in range(rB):
                    temp.append(A[i][k] * B[k][j])
                M[i].append(sum(temp))
#         print('row = {}, col = {}:\n{}'.format(rA, cB, M))
        return M
    except ValueError:
        raise ValueError("Matrix A's column number doesn't equal to Matrix b's row number")

def scaleRow(M, r, scale):
    try:
        if abs(scale) < 1e-10:
            raise ValueError
        row = M[r]
        for i in range(len(row)):
            row[i] = row[i] * scale
    except ValueError:
        raise ValueError('scale can not be zero.')

#  Gaussain Jordan  Ax = b

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    (r, c) = shape(A)
    (rb, cb) = shape(b)
    if r != rb:
        return None
    Ab = augmentMatrix(A, b)
    for j in range(0, c): #Ab
        maxrow = j
        for ii in range(j+1, r): # j
            if abs(Ab[ii][j]) > abs(Ab[maxrow][j]):
                maxrow = ii

        if abs(Ab[maxrow][j]) <= epsilon:#0
            return None
        swapRows(Ab, maxrow, j)#j

        #  j1
        coefficient = Ab[j][j]
        scaleRow(Ab, j, 1/coefficient)

        #  j 0
        for i in range(0, r):
            if i == j:
                continue
            coefficient = Ab[i][j]
            addScaledRow(Ab, i, j, -coefficient)

    matxRound(Ab, decPts)
    Mr = []
    for i in range(0, r):
        Mr.append([])
        Mr[i].append(Ab[i][-1])
    return Mr

def linearRegression(X,Y):
    XX = X
    YY = []
    for i in range(len(X)):
        YY.append([Y[i]])

    allOne = [[1.0]]*len(X)
    x = augmentMatrix(XX, allOne)
    xt = transpose(x)

    # bXY
    b = matxMultiply(xt, YY)

    # AXX
    A = matxMultiply(xt, x)

    h = gj_Solve(A, b)

    return [h[0][0], h[1][0], h[2][0]]
